SELECT Songs.Title, Songs.Id, 
KeySongs.SongId, KeySongs.KeyId,
Key.Id, Key.Name AS KeyName 
FROM Songs 
LEFT JOIN KeySongs ON Songs.Id = KeySongs.SongId
LEFT JOIN "Key" ON Key.Id = KeySongs.KeyId
